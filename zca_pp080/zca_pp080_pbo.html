*&---------------------------------------------------------------------*
*& Module INIT_ALV_0100 OUTPUT
*&---------------------------------------------------------------------*
MODULE INIT_ALV_0100 OUTPUT.

  IF GO_CONTAINER1 IS INITIAL.

*   Container, Grid 객체 생성
    PERFORM CREATE_SCREEN_OBJ USING GO_CONTAINER1
                                    GO_ALV_GRID1
                                    'CCON1'.
    PERFORM CREATE_SCREEN_OBJ USING GO_CONTAINER2
                                    GO_ALV_GRID2
                                    'CCON2'.
*   Layout 설정
    PERFORM SET_LAYOUT_CCON1.


*   Field Catalog 설정
    PERFORM SET_FIELD_CATALOG_CCON1.
*   Variant 설정
    PERFORM SET_VARIANT USING 110.
*   ALV 출력
    PERFORM SHOW_ALV USING GO_ALV_GRID1
                           GS_LAYO1
                           GT_FCAT
                           GT_DISPLAY_PPT030.

    PERFORM SET_FIELD_CATALOG_CCON2.
    PERFORM SET_VARIANT USING 120.
*   ALV 출력
    PERFORM SHOW_ALV USING GO_ALV_GRID2
                           GS_LAYO1
                           GT_FCAT
                           GT_DISPLAY_EBAN.


  ELSE.

    PERFORM REFRESH_GRID.

  ENDIF.

ENDMODULE.

MODULE FILL_DYNNR_0110 OUTPUT.

  CASE TABSTRIP-ACTIVETAB.

      DYNNR = '0110'.
    WHEN 'TAB2'.
      DYNNR = '0120'.
    WHEN OTHERS.
      TABSTRIP-ACTIVETAB = 'TAB1'.
      DYNNR = '0110'.
  ENDCASE.
ENDMODULE.
