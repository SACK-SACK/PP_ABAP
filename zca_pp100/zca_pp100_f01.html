FORM set_layo .
  CLEAR gs_layo.
  gs_layo-sel_mode = 'D'.       " 복수행 선택 + 행선택 컬럼 생성 + ANY 셀 선택 가능
  gs_layo-zebra = 'X'.
  gs_layo-cwidth_opt = 'X'.
ENDFORM.
*&---------------------------------------------------------------------*
*& Form SELECT_DATA
*&---------------------------------------------------------------------*
FORM select_data .
  SELECT *
    FROM zca_aufk
    INTO TABLE gt_display
     WHERE loekz EQ space
       AND werks EQ p_werks
       AND plnum EQ p_plnum.
  .
ENDFORM.
*&---------------------------------------------------------------------*
*& Form remove_timestemp
*&---------------------------------------------------------------------*
FORM remove_timestemp  TABLES pt_fcat STRUCTURE lvc_s_fcat
                       USING p_fieldname.

  CLEAR gs_fcat.
  gs_fcat-fieldname = p_fieldname.
  gs_fcat-no_out    = abap_on.
  APPEND gs_fcat TO pt_fcat.

ENDFORM.
