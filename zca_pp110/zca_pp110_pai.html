MODULE exit INPUT.
  CASE ok_code.
    WHEN 'EXIT'.
      LEAVE PROGRAM.
    WHEN 'CANC'.
      LEAVE TO SCREEN 0.
  ENDCASE.
ENDMODULE.

MODULE user_command_0100 INPUT.
  CASE ok_code.
    WHEN 'BACK'.
      LEAVE TO SCREEN 0.

    WHEN 'PRST'. "생산 시작 버튼
      PERFORM production_start.

    WHEN 'PODS'. "생산오더 조회 프로그램으로 이동
      PERFORM submit_to_po_program USING gs_display_100.

    WHEN 'PLDS'. "생산계획 조회 프로그램으로 이동
      PERFORM call_bdc_pl_program.
  ENDCASE.
ENDMODULE.
